13  import java.util.logging.Logger;
14  import javax.xml.transform.Transformer;
15  import javax.xml.transform.TransformerException;
16  import javax.xml.transform.TransformerFactory;
17  import javax.xml.transform.stream.StreamResult;
18  import javax.xml.transform.stream.StreamSource;

20  /**
21   *
22   * @author alok
23   */
24  public final class XMLToHTML {
25  
26      /**
27       * Unused Constructor.
28       */
29      private XMLToHTML() {
30      }
31  
32      /**
33       * Sample.
34       *
35       * @param args
36       */
37      public static void main(final String[] args) {
38          String path = "src/main/resources/checkstyle-report.html";
39          try (FileOutputStream outputStream = new FileOutputStream(path)) {
40              final TransformerFactory factory = TransformerFactory.newInstance();
41              path = "src/main/resources/checkstyle-report.xslt";
42              final StreamSource source = new StreamSource(Files.newBufferedReader(Paths.get(path)));
43              final Transformer transformer = factory.newTransformer(source);
44              final StreamResult outputTarget = new StreamResult(outputStream);
45              path = "/home/alok/Tools/netbeans8.1_workspace/SampleCheckstyle/target/checkstyle-result.xml";
46              final StreamSource xmlSource = new StreamSource(path);
47  
48              // Transform the XML Source to a Result
49              transformer.transform(xmlSource, outputTarget);
50          } catch (IOException | TransformerException ex) {
51              Logger.getLogger(XMLToHTML.class.getName()).log(Level.SEVERE, null, ex);
52          }
53      }
54  }
